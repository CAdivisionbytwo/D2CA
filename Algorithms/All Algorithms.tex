	$bs_i^t[3]\leftarrow null$\;

\IncMargin{1em} \begin{algorithm} [H]\label{111}
	\SetKwData{Left}{left}\SetKwData{This}{this}\SetKwData{Up}{up} \SetKwFunction{Union}{Union}
	\SetKwFunction{FindCompress}{FindCompress} 
	\SetKwInOut{Input}{input}\SetKwInOut{Output}{output}
	
	\Input{D2CA(x) and D2CA(y).\\
		T: number of levels (time steps).
		
	}	
	\Output{D2CA(z) = D2CA(x) + D2CA(y).} 
	
	$d\leftarrow Number\ of\ digit\ in\ x\ or\ y$\;
	$m\leftarrow Number\ of\ blocks\ in\ time\ step\ t$\;
	
	\For{$(t=0$ \KwTo $T-1)$}
	{   
		$bc\leftarrow \left\lfloor t/2 \right\rfloor+2$; // block counts\\
		$c\leftarrow 0$;\ //carry digit\\
		\If{$(t\ is\ even)$} {  
			
			\For{$(i=0$ \KwTo $bc-1)$}{
				
				$sum\leftarrow  (bs_{i}^t[1])_{x}+(bs_{i}^t[1])_{y}+c$;\\
				
				\If{$(sum>=10)$}{
					$(bs_{i}^t[1])_{z}\leftarrow sum-10$;\\
					$c\leftarrow 1$;
				}
				\Else{
					$(bs_{i}^t[1])_{z}\leftarrow sum$;\\
					$c\leftarrow 0$;
				}
			}
			\If{$(c==1)$}{
				$(bs_{i+1}^t[1])_{z}\leftarrow 1$;\\		
			}
			
		}
		
		
		
		\Else(//t is odd){
			
			
			
			\For{$(i=0$ \KwTo $bc-1)$}{
				
				$sum\leftarrow  (bs_{i}^t[2])_{x}+(bs_{i}^t[2])_{y}+c$;\\
				
				\If{$(sum>=10)$}{
					$(bs_{i}^t[2])_{z}\leftarrow sum-10$;\\
					$c\leftarrow 1$;
				}
				\Else{ 
					$(bs_{i}^t[2])_{z}\leftarrow sum$;\\
					$c\leftarrow 0$;
				}
				
				$sum\leftarrow  (bs_{i}^t[0])_{x}+(bs_{i}^t[0])_{y}+c$;\\
				
				\If{$(sum>4)$}{
					$(bs_{i}^t[0])_{z}\leftarrow\ sum- 5$;\\
					$c\leftarrow 5$;
				}
				\Else{
					$(bs_{i}^t[0])_{z}\leftarrow sum$;\\
					$c\leftarrow 0$;
				}
				
			}
			
			
			\If{$(c>0)$}{
				$(bs_{i+1}^t[2])_{z}\leftarrow c$;\\
				$(bs_{i+1}^t[0])_{z}\leftarrow 0$;\\
			}
			
		}	
		
	}

	\caption{Sum of D2CA(x) and D2CA(y)}\label{Code Geneator Procedure} \end{algorithm}\DecMargin{1em}\

	\SetKwData{Left}{left}\SetKwData{This}{this}\SetKwData{Up}{up} \SetKwFunction{Union}{Union}
	\SetKwFunction{FindCompress}{FindCompress} 
	\SetKwInOut{Input}{input}\SetKwInOut{Output}{output}
	
	\Input{x, a list of digits in number X.\\
		
	}	
	\Output{y, the quotients of X/2 .} 
	$y[0]\leftarrow  ((x[0]*5)\ mod\ 10)/10$\;	
	
	\For{$(i=0$ \KwTo $n)$}
	{   
		$d1\leftarrow x[i]$\;
		$d2\leftarrow x[i+1]$\;
		\If{$(d1\ is\ even)$}{ 
			\Switch{d2}{
				\Case{0,1}{
					$y[i]\leftarrow 0$
				}
				
				\Case{2,3}{
					$y[i]\leftarrow 1$
				}
				\Case{4,5}{
					$y[i]\leftarrow 2$
				}
				\Case{6,7}{
					$y[i]\leftarrow 3$
				}
				\Case{8,9}{
					$y[i]\leftarrow 4$
				}
			}
		}	
		\Else{
			\Switch{d2}{
				\Case{0,1}{
					$y[i]\leftarrow 5$
				}
				\Case{2,3}{
					$y[i]\leftarrow 6$
				}
				\Case{4,5}{
					$y[i]\leftarrow 7$
				}
				\Case{6,7}{
					$y[i]\leftarrow 8$
				}
				\Case{8,9}{
					$y[i]\leftarrow 9$
				}
			}
			
		}
		
	}	
	
	
	$y[n+1]\leftarrow decimal\ section\ of\ (x[n]*5)/10$\;	
	\caption{Division by Replacement)}\label{Code Geneator Procedure} \end{algorithm}\DecMargin{1em}\
